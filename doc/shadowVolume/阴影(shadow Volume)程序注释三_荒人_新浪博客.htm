    阴影(shadow Volume)程序注释三(2006-09-17 20:11:47)
      分类：游戏开发
    下面看如何利用上面得类，来绘制阴影。。。
    看仔细咯。。。
    bool CGraphics::OneFrameRender() {
     //clear any existing colour and depth information.
     //MUST CLEAR to BLACK (0,0,0) to stop it interfering with additive lighting
     pDev->Clear( 0, NULL, D3DCLEAR_TARGET | D3DCLEAR_ZBUFFER, D3DCOLOR_XRGB(0,0,0), 1.0f, 0 );
     pDev->BeginScene();
     //第一步：照常使用 z 缓冲区来描绘没有阴影的场景，关闭light
     if ( bAmbientLight ) {
      //colour buffer ON
      pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, FALSE );
      pDev->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_ONE );
      pDev->SetRenderState( D3DRS_DESTBLEND, D3DBLEND_ONE );
      //ambient lighting ON
      pDev->SetRenderState( D3DRS_LIGHTING, TRUE );
      pDev->SetRenderState( D3DRS_AMBIENT, AMB_LIGHT );
      pDev->LightEnable( 0, FALSE );
      pDev->LightEnable( 1, FALSE );
      pDev->LightEnable( 2, FALSE );
      pDev->LightEnable( 3, FALSE );
     } else { 
      //colour buffer OFF
      pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, TRUE );
      pDev->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_ZERO );
      pDev->SetRenderState( D3DRS_DESTBLEND, D3DBLEND_ZERO );
      pDev->SetRenderState( D3DRS_LIGHTING, FALSE ); //lighting OFF
     } //if(bAmbientLight)

     //depth buffer ON (write+test)
     pDev->SetRenderState( D3DRS_ZENABLE, TRUE );
     pDev->SetRenderState( D3DRS_ZWRITEENABLE, TRUE );
     pDev->SetRenderState( D3DRS_STENCILENABLE, FALSE ); //stencil buffer OFF

     //FIRST PASS: render scene to the depth buffer
     for ( int i = 0; i < 8; i++ ) {
      mshCaster[i]->Render( pDev ); //渲染整个场景
     }
     //OPTIMISED: all subsequent rendering is subject to this
     pDev->SetRenderState( D3DRS_ZWRITEENABLE, FALSE );

    //第二步：在模板缓冲区中将阴影标出来，帧缓冲区位于阴影中的部分模板值为零
    //colour buffer OFF
    //ambient lighting OFF
    //depth buffer ON (test ONLY)
    //stencil buffer OFF
    //for each light
    int iSVolIdx = 0;
    for ( int i = 0; i < iLightCount; i ++ ) {
    //if  enabled && casting shadows
     if ( lList[i]->bEnabled && lList[i]->bCastsShadows ) {
     //clear stencil buffer
     pDev->Clear( 0, NULL, D3DCLEAR_STENCIL, D3DCOLOR_XRGB(0,0,0), 1.0f, 0 );

     //turn OFF colour buffer
     pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, TRUE );
     pDev->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_ZERO );
     pDev->SetRenderState( D3DRS_DESTBLEND, D3DBLEND_ONE );
     //disable lighting (not needed for stencil writes!)
     pDev->SetRenderState( D3DRS_LIGHTING, FALSE );
     // z 测试被启用，而 z 写入被禁用，启用模板缓冲
             pDev->SetRenderState( D3DRS_STENCILENABLE, TRUE );
     pDev->SetRenderState( D3DRS_STENCILFUNC,  D3DCMP_ALWAYS );

     //render shadow volume
     //OPTIMISED: use support for 2-sided stencil
     //USE THE TRADITIONAL 2-PASS METHOD
     // 渲染阴影量的正面，每有一个像素通过 z 测试时，就将模板缓冲区增加一次 
             pDev->SetRenderState( D3DRS_STENCILPASS,  D3DSTENCILOP_INCR );
     pDev->SetRenderState( D3DRS_CULLMODE, D3DCULL_CCW ); //render front faces
     for ( int j = iSVolIdx; j < ( iSVolIdx + 4 ); j++ ) {
      pVols[j]->Render( pDev );
     }
             //阴影量的背面，每有一个像素通过 z 测试时，就将模板缓冲区减少一次
             pDev->SetRenderState( D3DRS_STENCILPASS,  D3DSTENCILOP_DECR );
     pDev->SetRenderState( D3DRS_CULLMODE, D3DCULL_CW ); //render back faces
     for ( int j = iSVolIdx; j < ( iSVolIdx + 4 ); j++ ) {
      pVols[j]->Render( pDev );
     }
     pDev->SetRenderState( D3DRS_CULLMODE, D3DCULL_CCW );
     }

     //Increment shadow volume index, slightly hack-ish but it'll do.
     //basically, there is no formula for the idx into pVols[] so we
     //just have to keep on counting...
     iSVolIdx += 4;

             // 第三步：最后，要实际描绘阴影，打开光照，整个场景会被一个 alpha 混色的多边形集充
             // 溢一遍，其中仅模板值非零的像素受到影响
     //alter stencil buffer
     pDev->SetRenderState( D3DRS_STENCILFUNC, D3DCMP_GREATER );
     //测试通过则保持原值，即阴影区不变，其他场景会被 alpha 混色的多边形集充溢一遍
     pDev->SetRenderState( D3DRS_STENCILPASS, D3DSTENCILOP_KEEP );

     //turn on CURRENT light, turn off all others
     pDev->SetRenderState( D3DRS_LIGHTING, TRUE );
     pDev->SetRenderState( D3DRS_AMBIENT, D3DCOLOR_XRGB(0,0,0) ); //关闭环境光
     pDev->LightEnable( 0, i == 0 ? TRUE : FALSE );
     pDev->LightEnable( 1, i == 1 ? TRUE : FALSE );
     pDev->LightEnable( 2, i == 2 ? TRUE : FALSE );
     pDev->LightEnable( 3, i == 3 ? TRUE : FALSE );
         
     //turn ON colour buffer
     pDev->SetRenderState( D3DRS_ALPHABLENDENABLE, TRUE );
     pDev->SetRenderState( D3DRS_SRCBLEND, D3DBLEND_ONE );
     pDev->SetRenderState( D3DRS_DESTBLEND, D3DBLEND_ONE );
         
     //render scene
     for ( int k = 0; k < 8; k++ ) {
      mshCaster[k]->Render( pDev );//渲染整个场景
     }

     //reset any necessary render states
     pDev->SetRenderState( D3DRS_STENCILENABLE, FALSE );
    } //if(enabled&&caster)
    } //for(each light)

     RenderLights( );
     //tell D3D we've finished sending geometry to be rendered.
     pDev->EndScene();
     pDev->Present(NULL,NULL,NULL,NULL);
     return true;
    } //OneFrameRender()
